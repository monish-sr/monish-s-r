MODULE_02   OPERATION

WORKING PRINCIPLE

⟳
RESET PHASE
  • Triggered when rst_i = 1
  • All FIFO slots cleared to 0
  • empty_o asserted, full_o deasserted
  • Both pointers and toggles reset
  • Synchronous on negedge clk
✎
WRITE OPERATION
  • wr_en_i = 1
  • If full_o = 0: write wdata_i → fifo[wr_pntr]
  • wr_pntr increments, wraps at depth−1
  • wr_tggle flips on wrap-around
  • If full_o = 1: overflow_o asserted
◈
READ OPERATION
  • rd_en_i = 1
  • If empty_o = 0: read fifo[rd_pntr] → rdata_o
  • rd_pntr increments, wraps at depth−1
  • rd_tggle flips on wrap-around
  • If empty_o = 1: underflow_o asserted
⊠
FULL DETECTION
  • Combinational always @(*)
  • wr_pntr == rd_pntr
  • wr_tggle != rd_tggle
  • Both conditions must be true
  • Toggle distinguishes full vs empty
⊡
EMPTY DETECTION
  • Combinational always @(*)
  • wr_pntr == rd_pntr
  • wr_tggle == rd_tggle
  • Both pointers same, same toggle
  • Asserted on reset by default
MODULE_03   TESTPLAN

DIRECTED TESTCASES

TC_01
FULL
Writes exactly depth items. Verifies full_o asserts after the last write slot is occupied.
Sequence: write(depth)
TC_02
EMPTY
Fills FIFO completely, then reads all entries. Verifies empty_o reasserts after all data is drained.
Sequence: write(depth) → read(depth)
TC_03
OVERFLOW
Attempts to write depth+1 items into a full FIFO. Verifies overflow_o is asserted on the extra write.
Sequence: write(depth+1)
TC_04
UNDERFLOW
Overflows first, then reads depth+1 times. Verifies underflow_o on the extra read from empty FIFO.
Sequence: write(depth+1) → read(depth+1)

module fifo #(parameter depth = 4, width = 3,
              ptr_width = $clog2(depth))
  (clk_i,rst_i,wr_en_i,rd_en_i,wdata_i,
   rdata_o,full_o,empty_o,overflow_o,underflow_o);

// Input and Output Declaration
input  clk_i,rst_i,wr_en_i,rd_en_i;
input  [width-1:0] wdata_i;
output reg [width-1:0] rdata_o;
output reg full_o,empty_o,overflow_o,underflow_o;

reg [ptr_width-1:0] wr_pntr, rd_pntr;
reg wr_tggle, rd_tggle;
reg [width-1:0] fifo [depth-1:0];
integer i;

// Toggles on Falling Edge
always@(negedge clk_i) begin
  if(rst_i) begin
    full_o <= 0;  empty_o <= 1;
    overflow_o <= 0;  underflow_o <= 0;
    rdata_o <= 0;
    {wr_pntr,rd_pntr,wr_tggle,rd_tggle} <= 0;
    for(i=0; i<depth; i=i+1) fifo[i] <= 0;
  end
  else begin
    if(wr_en_i) begin
      if (full_o) overflow_o <= 1;
      else begin
        fifo[wr_pntr] <= wdata_i;
        if(wr_pntr == depth-1) begin
          wr_tggle <= ~wr_tggle;  wr_pntr <= 0;
        end
        else wr_pntr <= wr_pntr + 1;
      end
    end
    if(rd_en_i) begin
      if(empty_o) underflow_o <= 1;
      else begin
        rdata_o <= fifo[rd_pntr];
        if(rd_pntr == depth-1) begin
          rd_tggle <= ~rd_tggle;  rd_pntr <= 0;
        end
        else rd_pntr <= rd_pntr + 1;
      end
    end
  end
end

// Combinational Full/Empty Detection
always@(*) begin
  empty_o = 0;  full_o = 0;
  if(wr_pntr==rd_pntr && wr_tggle==rd_tggle) empty_o = 1;
  if(wr_pntr==rd_pntr && wr_tggle!=rd_tggle) full_o  = 1;
end

endmodule
`include "fifo.v"

module tb #(parameter depth=4, width=3,
            ptr_width=$clog2(depth));

reg  clk_i,rst_i,wr_en_i,rd_en_i;
reg  [width-1:0] wdata_i;
wire [width-1:0] rdata_o;
wire full_o,empty_o,overflow_o,underflow_o;
integer i;
reg [8*10:0] testcase;

// DUT Instantiation
fifo #(.depth(depth),.width(width),.ptr_width(ptr_width))
  dut(.clk_i(clk_i), .rst_i(rst_i),
      .wr_en_i(wr_en_i), .rd_en_i(rd_en_i),
      .wdata_i(wdata_i), .rdata_o(rdata_o),
      .full_o(full_o), .empty_o(empty_o),
      .overflow_o(overflow_o), .underflow_o(underflow_o));

always #5 clk_i = ~clk_i;

// ── RESET TASK ──────────────────────────
task reset();
begin
  wdata_i=0; wr_en_i=0; rd_en_i=0; rst_i=1;
  repeat(2) @(negedge clk_i);
  rst_i=0;
end
endtask

// ── WRITE TASK ──────────────────────────
task write(input integer num);
begin
  for(i=0; i<num; i=i+1) begin
    @(negedge clk_i);
    wr_en_i = 1; wdata_i = $random();
  end
end
endtask

// ── READ TASK ───────────────────────────
task read(input integer num);
begin
  for(i=0; i<num; i=i+1) begin
    @(negedge clk_i);
    rd_en_i = 1;
  end
end
endtask

initial begin
  clk_i = 0;
  reset();
  if(!$value$plusargs("testcase=%s", testcase))
    $display("FAILED TO GET TESTCASE FROM THE USER");
  case(testcase)
    "FULL":      write(depth);
    "EMPTY":     begin write(depth); read(depth); end
    "OVERFLOW":  write(depth+1);
    "UNDERFLOW": begin write(depth+1); read(depth+1); end
    default:     $display("NO TC");
  endcase
end

initial #1000 $finish();
endmodule
