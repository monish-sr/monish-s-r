MODULE_02   OPERATION

WORKING PRINCIPLE

⟳
RESET PHASE
  • Triggered when rst_i = 1
  • All memory locations cleared to 0
  • ready_o deasserted
  • rdata_o cleared
  • Synchronous — acts on posedge clk
✎
WRITE OPERATION
  • valid_i = 1
  • wr_rd_i = 1
  • wdata_i → mem[addr_i]
  • ready_o asserted
  • Latched on posedge clk
◈
READ OPERATION
  • valid_i = 1
  • wr_rd_i = 0
  • mem[addr_i] → rdata_o
  • ready_o asserted
  • Sequential read of all locs in TB
⇄
HANDSHAKE LOGIC
  • ready_o = 1 when valid accepted
  • ready_o = 0 when valid = 0
  • TB waits on ready_o before next txn
  • Prevents data loss on back-to-back ops
  • Controlled flow without stalls

MODULE_04   VERIFICATION

TESTBENCH TASKS

⟳
task reset()
  • Initializes all signal to 0
  • Asserts rst_i = 1
  • Holds for 2 posedge clk cycles
  • Deasserts reset to begin DUT operation
✎
task write()
  • Iterates all 16 addresses (0→15)
  • Drives $random() data per location
  • Sets wr_rd_i=1, valid_i=1
  • Waits for ready_o before next write
  • Clears all signals post-loop
◈
task read()
  • Reads all 16 addresses sequentially
  • Sets wr_rd_i=0, valid_i=1
  • Captures rdata_o per address
  • Clears all signals post-loop
  • Simulation ends at #1000

// Memory Design

module memory #(parameter depth = 16, width = 4,
              addr_width = $clog2(depth))
  (clk_i, rst_i, wr_rd_i, valid_i,
   wdata_i, addr_i, ready_o, rdata_o);

// Signal declarations
input  clk_i, rst_i, wr_rd_i, valid_i;
input  [width-1:0]      wdata_i;
input  [addr_width-1:0] addr_i;
integer i;

output reg             ready_o;
output reg [width-1:0] rdata_o;
reg [width-1:0] mem [depth-1:0];

always@(posedge clk_i) begin
  if(rst_i) begin
    ready_o <= 0;
    rdata_o <= 0;
    for(i=0; i<depth; i=i+1) mem[i] <= 0;
  end
  else begin
    if(valid_i) begin
      ready_o <= 1;
      if(wr_rd_i) mem[addr_i] <= wdata_i;
      else        rdata_o   <= mem[addr_i];
    end
    else ready_o <= 0;
  end
end

endmodule
`include "memory_design.v"

module tb #(parameter width=4, depth=16,
            addr_width=$clog2(depth));

reg  clk_i,rst_i,valid_i,wr_rd_i;
reg  [width-1:0]      wdata_i;
reg  [addr_width-1:0] addr_i;
integer i;
wire                   ready_o;
wire [width-1:0]       rdata_o;

// DUT Instantiation
memory #(.width(width),.depth(depth),.addr_width(addr_width))
  dut (.clk_i(clk_i), .rst_i(rst_i),
       .valid_i(valid_i), .wr_rd_i(wr_rd_i),
       .wdata_i(wdata_i), .addr_i(addr_i),
       .ready_o(ready_o), .rdata_o(rdata_o));

always #5 clk_i = ~clk_i;   // 10ns period

// ── RESET TASK ──────────────────────────
task reset();
begin
  rst_i=1; wr_rd_i=0; wdata_i=0; valid_i=0; addr_i=0;
  repeat(2) @(posedge clk_i);
  rst_i=0;
end
endtask

// ── WRITE TASK ──────────────────────────
task write();
begin
  for(i=0; i<depth; i=i+1) begin
    @(posedge clk_i);
    wr_rd_i = 1; wdata_i = $random();
    addr_i  = i; valid_i = 1;
    wait(ready_o == 1);
  end
  @(posedge clk_i);
  {wr_rd_i,wdata_i,addr_i,valid_i} = 0;
end
endtask

// ── READ TASK ───────────────────────────
task read();
begin
  for(i=0; i<depth; i=i+1) begin
    @(posedge clk_i);
    wr_rd_i = 0; addr_i = i; valid_i = 1;
  end
  @(posedge clk_i);
  {wr_rd_i,addr_i,valid_i} = 0;
end
endtask

initial begin
  clk_i = 0;
  reset();   // PHASE 1
  write();   // PHASE 2
  read();    // PHASE 3
end

initial #1000 $finish();

endmodule
